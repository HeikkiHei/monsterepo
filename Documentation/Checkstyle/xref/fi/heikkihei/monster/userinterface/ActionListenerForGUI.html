3   import fi.heikkihei.monster.creature.Creature;
4   import fi.heikkihei.monster.generatestats.GenerateCreature;
5   import java.awt.event.*;
6   import java.io.FileNotFoundException;
7   import java.util.logging.Level;
8   import java.util.logging.Logger;
9   import javax.swing.*;
10  
11  /**
12   *
13   * @author HeikkiHei
14   */
15  public class ActionListenerForGUI implements ActionListener {
16  
17      private JButton generate;
18      private JTextArea output;
19      private JSlider setLevel;
20      private JRadioButton npcButton;
21      private JRadioButton monsterButton;
22      private GenerateCreature generateStats;
23  
24      /**
25       *
26       * @param generate nappi, jolla hahmo luodaan.
27       * @param output alue, johon tiedot tulostuvat.
28       * @param setLevel liuku, jolla määritetään hahmon taso.
29       * @param npcButton radionappi, jos halutaan npc.
30       * @param monsterButton radionappi, jos halutaan monster.
31       */
32      public ActionListenerForGUI(JButton generate, JTextArea output, JSlider setLevel, JRadioButton npcButton, JRadioButton monsterButton) {
33          this.generate = generate;
34          this.output = output;
35          this.setLevel = setLevel;
36          this.npcButton = npcButton;
37          this.monsterButton = monsterButton;
38  
39      }
40  
41      /**
42       * Kuunnellaan nappia Generate, lisätietoja radionapeista ja liuku'usta.
43       *
44       * @param ae kuunneltava event.
45       *
46       */
47      @Override
48      public void actionPerformed(ActionEvent ae) {
49          this.generateStats = new GenerateCreature(this.setLevel.getValue());
50          if (ae.getSource() == this.generate) {
51              if (this.npcButton.isSelected()) {              
52                  try {
53                      this.output.setText(this.generateStats.create());
54                  } catch (FileNotFoundException ex) {
55                      Logger.getLogger(ActionListenerForGUI.class.getName()).log(Level.SEVERE, null, ex);
56                  }
57              }
58              if (this.monsterButton.isSelected()) {
59                  try {
60                      this.output.setText(this.generateStats.create());
61                  } catch (FileNotFoundException ex) {
62                      Logger.getLogger(ActionListenerForGUI.class.getName()).log(Level.SEVERE, null, ex);
63                  }
64              }
65          }
66      }
67  
68  }
